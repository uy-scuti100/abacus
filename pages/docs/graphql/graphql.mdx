# Fetching Products Using GraphQL

## Using `graphql-request`

1. **Install the library**:

   ```bash js copy
   npm install graphql-request
   ```

2.**Query the data**:

    ```js copy
    import { gql, request } from "graphql-request";

    const storeId = "your-store-id";
    const baseUrl = "your base-url";
    const userId = "your-user-id";
    const apiKey = "your-api-key";

    // All This will be in your settings page when you logged in to the dashboard

    const PRODUCTS_QUERY = gql`
        query Products($store_id: uuid!, $user_id: uuid!) {
            products(
                where: { store_id: { _eq: $store_id }, user_id: { _eq: $user_id } }
            ) {
                id
                name
            }
        }
    `;

    async function fetchProducts(storeId, userId) {
        const variables = { store_id: storeId, user_id: userId };

        const headers = {
            "Content-Type": "application/json",
            apikey: apiKey,
        };

        try {
            const response = await request(baseUrl, PRODUCTS_QUERY, variables, headers);
            const data = response.products;
            console.log(data);

            // Example of mapping through products
            data.forEach((product) => {
                console.log(`Product ID: ${product.id}, Product Name: ${product.name}`);
            });

            // Using a single product in JSX
            const firstProduct = data[0];
            return (
                <div>
                    <h1>First Product</h1>
                    <p>ID: {firstProduct.id}</p>
                    <p>Name: {firstProduct.name}</p>
                </div>
            );
        } catch (error) {
            console.error("Error fetching products:", error);
        }
    }

    fetchProducts("your-store-id", "your-user-id");
    ```
