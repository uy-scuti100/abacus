# Fetching Products Using REST API

## Using `fetch`

1. **Construct the URL**:

   ```js copy
   const storeId = "your-store-id";This will be in your settings page when you logged in to the dashboard
   const baseUrl = "your base-url"
   const userId = "your-user-id";
   const apiKey = "your-api-key"; // This will be in your settings page when you logged in to the dashboard

   const fetchUrl = `${baseUrl}/products?store_id=eq.${storeId}&user_id=eq.${userId}`;
   ```

2. **Query the data**:

```js copy
async function fetchProducts() {
	try {
		const response = await fetch(fetchUrl, {
			headers: {
				"Content-Type": "application/json",
				apikey: apiKey,
				Authorization: `Bearer ${apiKey}`,
			},
		});

		if (!response.ok) {
			throw new Error("Network response was not ok");
		}

		const data = await response.json();
		console.log(data);

		// Example of mapping through products
		data.forEach((product) => {
			console.log(`Product ID: ${product.id}, Product Name: ${product.name}`);
		});

		// Using a single product in JSX
		const firstProduct = data[0];
		return (
			<div>
				<h1>First Product</h1>
				<p>ID: {firstProduct.id}</p>
				<p>Name: {firstProduct.name}</p>
			</div>
		);
	} catch (error) {
		console.error("Fetch error:", error);
	}
}

fetchProducts();
```
